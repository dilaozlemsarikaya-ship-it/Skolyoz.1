  <!-- TensorFlow.js ve Teachable Machine image library -->
  <script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs@latest/dist/tf.min.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/@teachablemachine/image@latest/dist/teachablemachine-image.min.js"></script>

  <script>
    // Ayarlar (eğer model klasörünü yerel koyduysan yolu böyle bırak)
    const MODEL_PATH = "model/model.json";       // <--- model klasörünü index.html ile aynı dizine koy
    const METADATA_PATH = "model/metadata.json";

    // Görüntü sınıf isimleri Teachable Machine ile eşleşmeli
    // Senin verdiğin tam Türkçe sınıf isimleri:
    const CLASS_LABELS = [
      "Normal Duruş",
      "Hafif Skolyozlu Duruş",
      "İleri Derece Skolyozlu Duruş"
    ];

    let model, maxPredictions;
    const inputEl = document.getElementById("input-image");
    const preview = document.getElementById("preview");
    const resultsDiv = document.getElementById("results");
    const mainResult = document.getElementById("main-result");
    const loading = document.getElementById("loading");
    const confidenceNote = document.getElementById("confidence-note");

    async function loadModel() {
      loading.style.display = "inline";
      try {
        model = await tmImage.load(MODEL_PATH, METADATA_PATH);
        maxPredictions = model.getTotalClasses();
        // Eğer metadata sınıf isimleri farklıysa bunları güvenlik amaçlı kontrol edebiliriz
        loading.style.display = "none";
      } catch (err) {
        loading.style.display = "none";
        alert("Model yüklenirken hata oluştu. Model klasörünün doğru dizinde ve 'model/model.json' adıyla bulunduğundan emin ol.");
        console.error(err);
      }
    }

    // Sayfa açıldığında modeli yükle
    loadModel();

    // Resim seçildiğinde otomatik preview + predict
    inputEl.addEventListener("change", async (evt) => {
      const file = evt.target.files && evt.target.files[0];
      if (!file) return;

      // preview
      const url = URL.createObjectURL(file);
      preview.src = url;

      // Kısa delay ile görüntü yüklendikten sonra predict et
      preview.onload = async () => {
        // Bekle model yüklendi mi
        if (!model) {
          // Eğer model henüz yüklenmediyse kullanıcıyı bilgilendir
          loading.style.display = "inline";
          await waitForModel();
          loading.style.display = "none";
        }
        await runPredict(preview);
        // serbest bırak
        URL.revokeObjectURL(url);
      };
    });

    // Model hazır olana kadar bekleyen yardımcı
    function waitForModel() {
      return new Promise((resolve) => {
        const t = setInterval(() => {
          if (model) { clearInterval(t); resolve(); }
        }, 200);
      });
    }

    async function runPredict(imageElement) {
      try {
        // Teachable Machine predict - doğrudan image element ile
        const prediction = await model.predict(imageElement);

        // prediction array uzunluğu maxPredictions
        // Build results list: her satır için yüzde
        // Map sınıfları kesin olarak CLASS_LABELS ile eşleştiriyoruz (güvenlik)
        resultsDiv.innerHTML = ""; // temizle

        // create an array of {label, prob}
        const items = prediction.map((p, i) => {
          const label = CLASS_LABELS[i] ?? p.className ?? ("Sınıf " + i);
          const prob = (p.probability ?? p.prob ?? 0);
          return { label, prob, index: i };
        });

        // show each with percentage
        items.forEach(it => {
          const percent = (it.prob * 100).toFixed(1) + "%";
          const wrapper = document.createElement("div");
          wrapper.className = "label";
          wrapper.innerHTML = `<strong>${it.label}</strong><span class="percent">${percent}</span>`;
          resultsDiv.appendChild(wrapper);
        });

        // highlight top result
        items.sort((a,b)=> b.prob - a.prob);
        const top = items[0];
        mainResult.innerText = `Tahmin: ${top.label} — %${(top.prob*100).toFixed(1)}`;
        confidenceNote.innerText = "Model çıktıları tahmini olasılıklardır. Klinik değerlendirme yerine geçmez.";

      } catch (err) {
        console.error("Predict hatası:", err);
        alert("Analiz sırasında hata oluştu. Konsolu kontrol et.");
      }
    }

    // Eğer istersen test için local bir örnek resmi otomatik yükleyebilirsin.
    // (Sunumda kullanma istersen kaldır)
